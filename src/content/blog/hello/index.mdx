#### Fourth level

Inline code looks like `pnpm build` or `getCollection('blog')`. A short TypeScript snippet:

```ts
type Post = {
  title: string
  date: Date
}

export function sortByDate(posts: Post[]) {
  return [...posts].sort((a, b) => b.date.getTime() - a.date.getTime())
}
```
